    <br>
    <img id="result">
    <script>
        function generate() {
            let prompt = document.getElementById('prompt').value;
            if (!prompt) return;
            let url = "https://image.pollinations.ai/prompt/" + encodeURIComponent(prompt);
            document.getElementById('result').src = url;
        }
    </script>
